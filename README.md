# sami-chatbot

## Install

```bash
npm install --save sami-chatbot
```

## Usage

```tsx
import React, { SamiChatBox } from 'react'

import SamiChatBox from 'sami-chatbot'
import 'sami-chatbot/dist/index.css'

class Example extends Component {
  render() {
    return <SamiChatBox />
  }
}
```
